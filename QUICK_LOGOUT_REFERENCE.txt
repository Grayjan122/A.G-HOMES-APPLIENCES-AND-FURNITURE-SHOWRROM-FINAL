🎯 ALL LOGOUT SCENARIOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

┌────────────────────────────────────────────────────────────────────┐
│  ACTION                      → RESULT                               │
├────────────────────────────────────────────────────────────────────┤
│  Close tab (X button)        → Instant logout, user offline         │
│  Close browser window        → Instant logout, user offline         │
│  Click back button (←)       → Instant logout, redirect to login    │
│  Inactive for 1 hour         → Warning + 60s countdown              │
│  Click "Logout" button       → Manual logout, user offline          │
│  Tab hidden for 1 hour       → Auto logout, user offline            │
└────────────────────────────────────────────────────────────────────┘


⚡ INSTANT LOGOUT (No Warning)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✓ Tab Close (X)
  └─→ Uses: sendBeacon API
  └─→ Event: beforeunload + pagehide
  └─→ Action: Set offline immediately
  └─→ Session: Cleared automatically

✓ Back Button (←)
  └─→ Uses: popstate event
  └─→ Event: Browser history navigation
  └─→ Action: Logout + redirect to login
  └─→ Session: Cleared automatically

✓ Browser Close
  └─→ Uses: sendBeacon API
  └─→ Event: beforeunload + pagehide
  └─→ Action: Set offline immediately
  └─→ Session: Cleared automatically


⏱️ DELAYED LOGOUT (With Warning)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✓ Inactivity Timeout
  └─→ Time: 1 hour (3600000 ms)
  └─→ Warning: Appears at 59 minutes
  └─→ Countdown: 60 seconds
  └─→ Options: "Stay Logged In" or "Logout Now"

🎬 EVENT FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Tab Close:
  User clicks X 
    └─→ beforeunload fires
        └─→ sendBeacon(offline)
            └─→ sessionStorage.clear()
                └─→ Tab closes
                    └─→ User offline in DB

Back Button:
  User clicks ← 
    └─→ popstate fires
        └─→ sendBeacon(offline)
            └─→ sessionStorage.clear()
                └─→ router.push('/')
                    └─→ Login page shown
                        └─→ User offline in DB
