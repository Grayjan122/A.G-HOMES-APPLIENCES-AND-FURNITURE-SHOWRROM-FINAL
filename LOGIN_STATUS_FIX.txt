🐛 PROBLEM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Issue Reported:
  "When I log in, my account status doesn't change to online"

Root Cause:
  The backend actStatus() function in login.php was using fetchAll()
  after an UPDATE query, which doesn't return any data. This made it
  difficult to verify if the status update was successful.


✅ SOLUTION IMPLEMENTED
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Two files were updated:

1. Frontend: app/page.js (OnlineState function)
   ✓ Added better logging
   ✓ Added success/failure checks
   ✓ Added console messages for debugging
   ✓ Returns response data for verification

2. Backend: capstone-api/api/login.php (actStatus function)
   ✓ Removed incorrect fetchAll() call
   ✓ Added rowCount() check to verify update
   ✓ Returns proper success/failure response
   ✓ Includes affected rows count


📝 WHAT CHANGED
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

BEFORE (login.php - actStatus function):
┌────────────────────────────────────────────────────┐
│ function actStatus($json){                         │
│     include 'conn.php';                            │
│     $json = json_decode($json, true);              │
│     $sql = "UPDATE `account`                       │
│             SET `active_status`= :statusState      │
│             WHERE account_id = :userID";           │
│                                                    │
│     $stmt = $conn->prepare($sql);                  │
│     $stmt->bindParam(':userID', $json['userID']);  │
│     $stmt->bindParam(':statusState',               │
│                      $json['state']);              │
│     $stmt->execute();                              │
│     $returnValue = $stmt->fetchAll(...);  ❌       │
│     return json_encode($returnValue);              │
│ }                                                  │
└────────────────────────────────────────────────────┘
Problem: fetchAll() doesn't return data for UPDATE queries!


AFTER (login.php - actStatus function):
┌────────────────────────────────────────────────────┐
│ function actStatus($json){                         │
│     include 'conn.php';                            │
│     $json = json_decode($json, true);              │
│     $sql = "UPDATE `account`                       │
│             SET `active_status`= :statusState      │
│             WHERE account_id = :userID";           │
│                                                    │
│     $stmt = $conn->prepare($sql);                  │
│     $stmt->bindParam(':userID', $json['userID']);  │
│     $stmt->bindParam(':statusState',               │
│                      $json['state']);              │
│     $stmt->execute();                              │
│                                                    │
│     $rowCount = $stmt->rowCount();  ✅             │
│                                                    │
│     if ($rowCount > 0) {                           │
│         $returnValue = [                           │
│             'success' => true,                     │
│             'message' => 'Status updated',         │
│             'status' => $json['state'],            │
│             'user_id' => $json['userID'],          │
│             'rows_affected' => $rowCount           │
│         ];                                         │
│     } else {                                       │
│         $returnValue = [                           │
│             'success' => false,                    │
│             'message' => 'No rows updated',        │
│             'status' => $json['state'],            │
│             'user_id' => $json['userID']           │
│         ];                                         │
│     }                                              │
│                                                    │
│     return json_encode($returnValue);              │
│ }                                                  │
└────────────────────────────────────────────────────┘
Solution: Use rowCount() and return proper response!
